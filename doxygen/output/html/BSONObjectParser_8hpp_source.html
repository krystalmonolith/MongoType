446  virtual void parseElementRecursive(const BSONElement& element, string& key, int elementIndex=0, int elementCount=1, int arrayIndex = -1, int arrayCount = 0) {
447  BSONType btype = element.type();
448  switch (btype) {
449  case BSONType::Object:
450  {
451  const BSONObj& bobj = element.Obj();
452  string k(element.fieldName());
453  parseObjectRecursive(bobj, k, elementIndex, elementCount, arrayIndex, arrayCount);
454  }
455  break;
456  case BSONType::Array:
457  {
458  stack.push(BSONParserStackItem::ItemType::ARRAY, element, key, elementIndex, elementCount, arrayIndex, arrayCount);
459  std::vector<BSONElement> elementArray = element.Array();
460  int elementArrayCount = elementArray.size();
461  visitor.onArrayStart(stack);
462  int elementArrayIndex = 0;
463  for (BSONElement e : elementArray) {
464  string k(e.fieldName());
465  parseElementRecursive(e, k, elementIndex, elementCount, elementArrayIndex++, elementArrayCount);
466  }
467  visitor.onArrayEnd(stack);
468  stack.drop();
469  }
470  break;
471  default:
472  {
473  stack.push(BSONParserStackItem::ItemType::ELEMENT, element, key, elementIndex, elementCount, arrayIndex, arrayCount);
474  visitor.onElement(stack);
475  stack.drop();
476  }
477  break;
478  }
